001 import java.util.HashMap; 
002import java.util.Set;
003public class Room
004{
005    //Définition des attibuts
006
007    private String aDescription;
008    private HashMap<String, Room> aExits;
009    private String aImageName;
010    private ItemList aItems;
011    //Constructeurs
012    /**
013     * Constructeur de la piece
014     * @param pD Description du lieu
015     * @param pI lien de l'image
016     */
017    public Room(final String pD,final String pI)
018    {
019        this.aDescription = pD;  
020        this.aExits = new HashMap<String, Room>();
021        this.aItems = new ItemList();
022        this.aImageName = pI;
023    }//room    
024    
025    //Accesseurs
026    /**
027     * Retourne la description du lieu
028     * @return Description
029     */
030    public String getDescription(){ return this.aDescription; }//GetDescription
031
032    /**
033     * Retourne la desctiption longue du lieu actuel, contenant la description du lieu et les sorties 
034     * @return Description longue
035     */
036    public String getLongDescription()
037    {
038        if( this.aItems.isEmpty())
039            return "Vous êtes " + this.aDescription + ".\n" + this.getExitString();
040        else 
041            return "Vous etes " + this.aDescription + ".\n" + this.getItemString() + ".\n" + this.getExitString(); 
042    }
043    
044    /**
045     * Return a string describing the room's image name
046     * @return No mdel'image
047     */
048    public String getImageName(){return aImageName;}
049        
050    /**
051     * retourne la sortie correspondante
052     * @param pD Donne la piece dans la direction de ce parametre
053     * @return sortie dans la direction
054     */
055    public Room getExit(final String pD  )
056    {
057
058        return this.aExits.get(pD);
059
060    }//GetExit
061
062    /**
063     * retourne les différentes sorties de la piece actuelle
064     * @return Les sorties
065     */
066    public String getExitString()
067    {
068        String vReturnString = "Les sorties sont:";
069        Set<String> vKeys = this.aExits.keySet();
070        for(String vExit : vKeys)
071        {
072            vReturnString += ", " + vExit;
073        }
074
075        return vReturnString;
076
077    }//getexitstring
078    
079    /**
080     * Récupere tous les objets dans la pièce
081     * @return Chaine de caractères avec tous les objets dedans
082     */
083    public String getItemString()
084    {
085       return "Les objets sont :" + this.aItems.toStringDetail();
086
087    }
088    //Mutateurs
089    /**
090     * Définit la sortie de la piece atuelle
091     * @param pD direction relative
092     * @param pV piece voisine dans cette direction
093     */
094    public void setExit(final String pD, final Room pV)
095    {
096        this.aExits.put(pD, pV);
097
098    }//setExit
099    //Methodes
100    /**
101     * retourne l'objet associé à un nom
102     * @param pName nom de l'objet
103     * @return objet en question ( peut etre nul)
104     */
105    public Item getItem(final String pName)
106    {
107        Item vItem = this.aItems.getItem(pName);
108        return vItem;        
109    }
110    
111    /**
112     * Ajoute un objet dans la pièce
113     * @param pI objet en question
114     */
115    public void addItem(final Item pI)
116    {
117        this.aItems.addItem(pI);
118    }
119    
120    /**
121     * Enleve un objet dans la pièce
122     * @param pI objet en question
123     */
124    public void removeItem(final String pN)
125    {
126        this.aItems.removeItem(pN);
127    }
128} // Room
