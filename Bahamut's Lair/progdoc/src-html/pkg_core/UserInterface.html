022/**
023 * This class implements a simple graphical user interface with a text entry
024 * area, a text output area and an optional image.
025 *

029public class UserInterface implements ActionListener
030{
031    private GameEngine aEngine;

035    private JTextArea  aLog;

053    /**
054     * Construct a UserInterface. As a parameter, a Game Engine
055     * (an object processing and executing the game commands) is
056     * needed.
057     * 
058     * @param pGameEngine  The GameEngine object implementing the game logic.
059     */
060    public UserInterface( final GameEngine pGameEngine )
061    {
062        this.aEngine = pGameEngine;
063        //Ordre important
064        this.aButtonList = new ArrayList<JButton>(10);
065        this.createGUI();
066        this.aParser = new Parser();
067    } // UserInterface(.)
068
069    /**
070     * Print out some text into the text area.
071     * @param pText texte à afficher
072     */
073    public void print( final String pText )
074    {
075        this.aLog.append( pText );
076        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
077    } // print(.)
078
079    /**
080     * Print out some text into the text area, followed by a line break.
081     * @param pText texte à afficher
082     */
083    public void println( final String pText )
084    {
085        this.print( pText + "\n" );
086    } // println(.)
087
088    /**
089     * Show an image file in the interface.
090     * @param pImageName "Lien" de l'image
091     */
092    public void showImage( final String pImageName )
093    {
094        URL vImageURL = this.getClass().getClassLoader().getResource( pImageName );
095        if ( vImageURL == null )
096            System.out.println( "image not found" );
097        else {
098            ImageIcon vIcon = new ImageIcon( vImageURL );
099            this.aImage.setIcon( vIcon );
100            this.aMyFrame.pack();
101        }
102    } // showImage(.)
103
104    /**
105     * Enable or disable input in the input field.
106     * @param pOnOff bolléen pour activer ou non
107     */
108    public void enable( final boolean pOnOff )
109    {
110        this.aEntryField.setEditable( pOnOff );
111        
112        for(JButton vB : this.aButtonList)
113            vB.setEnabled(pOnOff);
114        if ( ! pOnOff )
115            this.aEntryField.getCaret().setBlinkRate( 0 );
116 
117    } // enable(.)
118

216    /**
217     * Actionlistener interface for entry textfield.
218     * @param pE Action récupérée par les listener
219     */
220    public void actionPerformed( final ActionEvent pE ) 
221    {   
222        
223        //Remplacable en vrifiant que la classe d'éntrée est un bouton et e nrécupérant ce texte pour la commande.
224        
225        if(pE.getSource().getClass() == this.aButtonHarakiri.getClass())
226            this.aEngine.interpretCommand(this.aParser.getCommand(pE.getActionCommand()));
227        else
228            this.processCommand();
229        
230        
231        
232    } // actionPerformed(.)
233
234    /**
235     * A command has been entered. Read the command and do whatever is 
236     * necessary to process it.
237     */
238    private void processCommand()
239    {
240        String vInput = this.aEntryField.getText();
241        this.aEntryField.setText( "" );
242
243        this.aEngine.interpretCommand( this.aParser.getCommand(vInput) );
244    } // processCommand()
245} // UserInterface
