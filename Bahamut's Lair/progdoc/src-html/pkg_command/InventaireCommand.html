001package pkg_command;
002import pkg_core.GameEngine;
003import pkg_mechanics.ItemList;
004import pkg_mechanics.Arme;
005import pkg_mechanics.Armor;
006
007/**
008 * Commande permettant d'afficher l'inventaire
009 *
010 * @author Quentin GARRIDO
011 * @version 1.0
012 */
013public class InventaireCommand extends Command
014{
015    private CommandWords aCommandWords;
016    /**
017     * Constructeur de la classe InventaireCommand
018     * @param pCommandWords mot de commande associé
019     */
020    public InventaireCommand(final CommandWords pCommandWords)
021    {
022        super(null);
023        this.aCommandWords = pCommandWords;
024    }//Constructeur()
025    /** 
026     * permet d'afficher l'inventaire
027     * @param pGameEngine GameEngine qui éxécutera la commande
028     */
029    @Override
030    public void execute(final GameEngine pGameEngine)
031    {
032
033        String vName = super.getSecondWord(); 
034        ItemList vInventory = pGameEngine.getPlayer().getInventory();
035        if (vInventory.isEmpty())
036            pGameEngine.getGUI().println("Votre inventaire est vide"+ "\n" + pGameEngine.getPlayer().getEquipment());
037        else
038        {
039            String vString = "Dans voter inventaire se trouvent : "+ vInventory.toString() + " pour un poids total de " + pGameEngine.getPlayer().getCurrentWeight()+ "\n" + pGameEngine.getPlayer().getEquipment();
040            pGameEngine.getGUI().println(vString);
041        }
042    }//execute
043}
