001package pkg_command;
002import pkg_core.GameEngine;
003
004
005/**
006 * Commande permettant d'afficher l'aide
007 *
008 * @author Quentin GARRIDO
009 * @version 1.0
010 */
011public class HelpCommand extends Command 
012{
013    private CommandWords aCommandWords;
014    /**
015     * Constructeur de la classe HelpCommand
016     * @param pCommandWords mot de commande associé
017     */
018    public HelpCommand(final CommandWords pCommandWords)
019    {
020        super(null);
021        this.aCommandWords = pCommandWords;
022    }//Constructeur()
023    /** 
024     * affiche le message d'aide
025     * @param pGameEngine GameEngine qui éxécutera la commande
026     */
027    @Override
028    public void execute(final GameEngine pGameEngine)
029    {
030        pGameEngine.getGUI().println("Vous êtes seul et perdu. \nVous vous trouvez dans la montagne sacrée du Bahamut.\n");
031        pGameEngine.getGUI().println("Your command words are : " + this.aCommandWords.getCommandList() + "\n");
032    }//execute
033}
