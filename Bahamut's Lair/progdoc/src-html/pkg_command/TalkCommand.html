001package pkg_command;
002import pkg_core.GameEngine;
003import pkg_mechanics.ItemList;
004
005/**
006 * Commande permettant de parler à un npc
007 *
008 * @author Quentin GARRIDO
009 * @version 1.0
010 */
011public class TalkCommand extends Command
012{
013    private CommandWords aCommandWords;
014    /**
015     * Constructeur de la classe TalkCommand
016     * @param pCommandWords mot de commande associé
017     */
018    public TalkCommand(final CommandWords pCommandWords)
019    {
020        super(null);
021        this.aCommandWords = pCommandWords;
022    }//Constructeur()
023    /** 
024     * Permet de parler à un NPC si il est dans la pièce
025     * @param pGameEngine GameEngine qui éxécutera la commande
026     */
027    @Override
028    public void execute(final GameEngine pGameEngine)
029    {
030        if(super.hasSecondWord())
031            if(pGameEngine.getPlayer().getCurrentRoom().getNPC(super.getSecondWord()) != null)
032                pGameEngine.getGUI().println(  pGameEngine.getPlayer().getCurrentRoom().getNPCDialogue(super.getSecondWord()));
033            else
034                pGameEngine.getGUI().println("Ce personnage n'est pas dans la pièce !");
035        else
036            pGameEngine.getGUI().println("A qui voulez vous parler ?");
037    }//execute
038}
