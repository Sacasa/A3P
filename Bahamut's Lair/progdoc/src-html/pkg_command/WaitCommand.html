001package pkg_command;
002import pkg_core.GameEngine;
003import java.util.ArrayList;
004import pkg_mechanics.Room;
005
006/**
007 * Commande permettant d'attendre ( surtout pour le bahamut)
008 *
009 * @author Quentin GARRIDO
010 * @version 1.0
011 */
012public class WaitCommand extends Command 
013{
014    private CommandWords aCommandWords;
015    /**
016     * Constructeur de la classe WaitCommand
017     * @param pCommandWords mot de commande associé
018     */
019    public WaitCommand(final CommandWords pCommandWords)
020    {
021        super(null);
022        this.aCommandWords = pCommandWords;
023    }//Constructeur()
024    /** 
025     * permet de faire passer un "tour", ce qui sert à faire bouger les pnj sans rien faire d'autre majoritairement
026     * Mais mettra aussi le bahamut dans la salle 16(cf plan)
027     * @param pGameEngine GameEngine qui éxécutera la commande
028     */
029    @Override
030    public void execute(final GameEngine pGameEngine)
031    {
032        ArrayList<Room> vRoomList = pGameEngine.getRoomList();
033        if(super.hasSecondWord())
034        {
035            pGameEngine.getGUI().println("Attendre quoi ?");            
036        }
037        else
038        {        
039            pGameEngine.getBahamut().setCurrentRoom(vRoomList.get(16));
040            pGameEngine.getPlayer().changeRoom(pGameEngine.getPlayer().getCurrentRoom());
041            pGameEngine.isBahamutIn();
042        }
043        
044    }//execute
045}
