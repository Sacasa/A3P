001package pkg_mechanics;
002
003/**
004 * LE beamer est un objet permettant de se téléporter entre les salles
005 *
006 * @author Quentin Garrido
007 * @version 1.0
008 */
009public class Beamer extends Item
010{
011    private Room aChargedRoom;
012
013    /**
014     * Constructor for objects of class Beamer
015     */
016    public Beamer()
017    {
018        super("beamer","Un beamer, objet antique permettant de se téléporter", 5,false);
019
020    }
021
022    /**
023     * permet de charger la pièce dans la pièce courante
024     */
025    public void charge()
026    {
027        this.aChargedRoom = this.getCurrentRoom();
028
029    }
030
031    /**
032     * permet de se téléporter dans la pièce en mémoire
033     */
034    public void fire()
035    {
036        super.getCarrier().changeRoom(aChargedRoom);
037    }
038
039    /**
040     * récupère la pièce actuelle du porteur
041     * @return pièce actuelle
042     */
043    public Room getCurrentRoom()
044    {
045        return super.getCarrier().getCurrentRoom();
046    }
047
048    /**
049     * récupère la pièce en mémoire
050     * @return pièce en mémoire
051     */
052    public Room getChargedRoom()
053    {
054        return this.aChargedRoom;
055    }
056}
