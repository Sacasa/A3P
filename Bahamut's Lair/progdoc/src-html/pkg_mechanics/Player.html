015{
016    // variables d'instance - remplacez l'exemple qui suit par le vôtre

027    /**
028     * Constructeur de Player
029     * @param pN nom du player
030     * @param pTW poids total du joueur
031     * @param pR pièce initiale du joueur
032     */
033    public Player(final String pN, final int pTW,final Room pR)
034    {
035        this.aNom = pN;
036        this.aMaxWeight = pTW;
037        this.aCurrentWeight = 0;
038        this.aCurrentRoom = pR;
039        this.aInventory = new ItemList();
040        this.aArme = new Arme();
041        this.aArmor = new Armor();
042        this.aHP = 20.0;
043    }//player
044
045    /**
046     * ajoute du poids au maximum du joueur
047     * @param pI poids à ajouter(ou enlever)
048     */
049    public void addMaxWeight(final int pI)
050    {
051        this.aMaxWeight += pI;
052
053    }
054
055    /**
056     * récupère le poids maximumu du joueur
057     * @return poids max
058     */
059    public int getMaxWeight()

061        return this.aMaxWeight;
062    }
063    
064    public void addHP(final double pHP)
065    {
066        this.aHP += pHP;
067    }

069    /**
070     * retourne l'inventaire du joueur

086    /**
087     * Ajoute/enleve du poids actuel au joueur
088     * @param pI poids à ajouter/enlever

090    public void addCurrentWeight(final int pI)
091    {
092        this.aCurrentWeight += pI; 
093    }
094
095    /**
096     * Récupère le poids actuel du joueur

102
103    }

109    public Room getCurrentRoom()

111        return this.aCurrentRoom;
112
113    }//getCurrentRoom
114
115    /**
116     * Déplace l'utilisateur dans la piece donnée
117     * @param pRoom pièce où aller
118     */
119    public void changeRoom(final Room pRoom)
120    {
121        this.aCurrentRoom = pRoom; 
122        this.aGui.println(this.aCurrentRoom.getLongDescription());
123        if(this.aCurrentRoom.getImageName() != null)
124            this.aGui.showImage(this.aCurrentRoom.getImageName());  
125    }//ChangeRoom
126
127    /**
128     * permet de savoir si un objet est porté ou non
129     * @param pS nom de l'objet
130     */
131    public boolean isCarried(final String pS)
132    {
133        if (this.aInventory.getItem(pS) != null)
134            return true;
135        else
136            return false;
137
138    }

178     * @return statistique d'armure du joueur

183    }

185     * @return arme et armure du joueur sous forme de texte
186     */
187    public String getEquipment()
188    {
189        return "Votre arme est : " + this.aArme.toString() + "\n" + "Votre armure est : " + this.aArmor.toString() ;
190    }
191    /**
192     * Calcule et affiche la vie du joueur, ainsi que sa mort potentielle
193     */
194    public void status()
195    {
196        if(this.aHP > 0.0)
197           this.aGui.println("Vous avez " + this.aHP +" pv.");
198        else
199            {
200            this.aGui.println("Vous êtes mort.");  
201            this.aGui.showImage("./Images/Mort.jpg");
202            this.aGui.enable(false);
203            }
