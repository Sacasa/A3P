001package pkg_mechanics;

003import java.util.ArrayList;
004/**
005 * Sert a créer des NPC se déplaçant
006 *
007 * @author Quentin Garrido
008 * @version 1.0
009 */
010public class MovingNPC extends NPC
011{
012    // instance variables - replace the example below with your own
013    
014    private RoomRandomizer aRoomRandomizer;
015
016    /**
017     * Constructor for objects of class MovingNPC
018     * @param pD liste des dialogues du personnage
019     * @param pN Nom du personnage
020     * @param pAR Liste de salles où se déplacer
021     * @param pCR Piece initiale(non aléatoire)
022     */
023    public MovingNPC(final ArrayList<String> pD, final String pN,final double pHP, final int pDegats, final int pArmure, final boolean pK, final Item pLoot, final Room pCR, final ArrayList<Room> pAR)
024    {
025        super(pD,pN,pHP,pDegats,pArmure,pK,pLoot,pCR);
026        this.aRoomRandomizer = new RoomRandomizer(pAR);
027
028    }//MovingNPC
029
030    /**
031    *déplace le personnage dans une nouvelle pièce aléatoire
032    */
033    public void move()
034    {   
035        super.getCurrentRoom().getNPCList().remove(super.getName());
036        super.setCurrentRoom(this.aRoomRandomizer.getRandomRoom());    
037        super.getCurrentRoom().getNPCList().put(super.getName(), this);
038    }//move
039
040}
