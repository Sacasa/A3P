001package pkg_mechanics;
002
003import java.util.ArrayList;
004/**
005 * Room nous téléportant aléatoirement à sa sortie.
006 * 
007 * @author Quentin Garrido
008 * @version 1.0
009 */
010public class TransporterRoom extends Room
011{
012    private RoomRandomizer aRandomizer;  
013    private Integer aAleaNb;
014
015    /**
016     * Constructor for objects of class TransporterRoom
017     * @param pD Description de la pièce
018     * @param pI image de la pièce
019     * @param pR Liste de pièces aléatoires dans lesquelles choisir
020     */
021    public TransporterRoom(final String pD, final String pI, final ArrayList<Room> pR)
022    {
023        super(pD,pI);
024        aRandomizer = new RoomRandomizer(pR);
025
026    }
027    
028    /**
029     * définit la valeur de la pièce ou aller(blocage aléatoire)
030     * @param pI
031     */
032    public void setAlea(final Integer pI)
033    {
034        this.aAleaNb = pI;
035    }
036
037    /**
038     * Récupère la sortie de la pièce 
039     * @param pE direction de la sortie(ici inutile)
040     * @return pièce où aller 
041     */    
042    @Override
043    public Room getExit(String pE)
044    {
045        if((Integer)this.aAleaNb != null && aAleaNb < this.aRandomizer.getRandomRoomList().size()-1)
046            return this.aRandomizer.getRoom(this.aAleaNb);
047
048        //En cas d'erreur ou de comportement aléatoire :
049        return aRandomizer.getRandomRoom();
050
051    }
052
053
054}
